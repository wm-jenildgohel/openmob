<!-- Hub HTTP API contract that the MCP server proxies to -->

Hub API Base: http://127.0.0.1:8686/api/v1

| MCP Tool | HTTP Method | Hub Endpoint | Request Body | Response |
|----------|-------------|-------------|--------------|----------|
| list_devices | GET | /devices | - | [{id, serial, model, manufacturer, osVersion, sdkVersion, screenWidth, screenHeight, batteryLevel, batteryStatus, connectionType, status, bridgeActive, platform, deviceType}] |
| get_screenshot | GET | /devices/:id/screenshot | - | {screenshot: "base64...", width: number, height: number} |
| get_ui_tree | GET | /devices/:id/ui-tree?text=&visible= | - | {nodes: [{index, text, className, resourceId, contentDesc, bounds: {left,top,right,bottom,centerX,centerY}, visible}]} |
| tap | POST | /devices/:id/tap | {x, y} or {index} | {success: boolean, error?: string} |
| type_text | POST | /devices/:id/type | {text: string} | {success: boolean, error?: string} |
| swipe | POST | /devices/:id/swipe | {x1, y1, x2, y2, duration?: number} | {success: boolean, error?: string} |
| launch_app | POST | /devices/:id/launch | {package: string} | {success: boolean, error?: string} |
| terminate_app | POST | /devices/:id/terminate | {package: string} | {success: boolean, error?: string} |
| press_button | POST | /devices/:id/keyevent | {keyCode: number} | {success: boolean, error?: string} |
| go_home | POST | /devices/:id/keyevent | {keyCode: 3} | {success: boolean, error?: string} |
| open_url | POST | /devices/:id/open-url | {url: string} | {success: boolean, error?: string} |

MCP Image content type for screenshots:
```typescript
{ type: "image" as const, data: string /* base64 */, mimeType: "image/png" }
```

MCP Text content type for all other responses:
```typescript
{ type: "text", text: string /* JSON stringified */ }
```

   ```typescript
   const HUB_URL = process.env.OPENMOB_HUB_URL || "http://127.0.0.1:8686/api/v1";

   export async function hubGet<T = unknown>(path: string): Promise<T> {
     const url = `${HUB_URL}${path}`;
     const res = await fetch(url);
     if (!res.ok) {
       const body = await res.text();
       throw new Error(`Hub API error ${res.status}: ${body}`);
     }
     return res.json() as Promise<T>;
   }

   export async function hubPost<T = unknown>(path: string, body: Record<string, unknown>): Promise<T> {
     const url = `${HUB_URL}${path}`;
     const res = await fetch(url, {
       method: "POST",
       headers: { "Content-Type": "application/json" },
       body: JSON.stringify(body),
     });
     if (!res.ok) {
       const body_text = await res.text();
       throw new Error(`Hub API error ${res.status}: ${body_text}`);
     }
     return res.json() as Promise<T>;
   }
   ```

    import { StdioServerTransport } from "@modelcontextprotocol/sdk/server/stdio.js";
    // Import all register functions
    import { registerListDevices } from "./tools/list-devices.js";
    import { registerGetScreenshot } from "./tools/screenshot.js";
    import { registerGetUiTree } from "./tools/ui-tree.js";
    import { registerTap } from "./tools/tap.js";

    import { registerSwipe } from "./tools/swipe.js";
    import { registerLaunchApp } from "./tools/launch-app.js";
    import { registerTerminateApp } from "./tools/terminate-app.js";
    import { registerPressButton } from "./tools/press-button.js";
    import { registerGoHome } from "./tools/go-home.js";
    import { registerOpenUrl } from "./tools/open-url.js";

    const server = new McpServer({
      name: "openmob",
      version: "1.0.0",
    });

    // Register all tools
    registerListDevices(server);
    registerGetScreenshot(server);
    registerGetUiTree(server);
    registerTap(server);
    registerTypeText(server);
    registerSwipe(server);
    registerLaunchApp(server);
    registerTerminateApp(server);
    registerPressButton(server);
    registerGoHome(server);
    registerOpenUrl(server);

    async function main() {
      const transport = new StdioServerTransport();
      await server.connect(transport);
      console.error("OpenMob MCP Server running on stdio");
    }

    main().catch((error) => {
      console.error("Fatal error:", error);
      process.exit(1);
    });
