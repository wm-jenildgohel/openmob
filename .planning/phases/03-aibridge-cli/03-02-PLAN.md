  <action>
    1. Create internal/bridge/busy.go:
       - Package bridge
       - type BusyDetector struct:
         - mu sync.RWMutex
         - idle bool (starts false)
         - lastOutput time.Time
         - pattern *regexp.Regexp (busy pattern -- when matched, agent is BUSY)
         - onIdle func() (callback when transitioning busy->idle)
         - idleTimeout time.Duration (default 500ms)
         - tickRate time.Duration (default 100ms)
       - func NewBusyDetector(pattern *regexp.Regexp, idleTimeout time.Duration, onIdle func()) *BusyDetector
         - If idleTimeout is 0, default to 500ms
         - tickRate always 100ms
       - func (d *BusyDetector) ProcessLine(line string):
         - Lock mu
         - If pattern != nil and pattern matches line: set idle=false, update lastOutput
         - If pattern == nil: always set idle=false, update lastOutput (any output = busy)
         - Unlock
       - func (d *BusyDetector) IsIdle() bool -- read lock, return idle
       - func (d *BusyDetector) Run(ctx context.Context):
         - Creates ticker at tickRate
         - On each tick: lock, check if !idle && time.Since(lastOutput) > idleTimeout
         - If so: set idle=true, unlock, call onIdle()
         - Otherwise: unlock
         - Exit on ctx.Done()

    2. Create internal/bridge/queue.go:
       - Package bridge
       - const MaxQueueSize = 100
       - var ErrQueueFull = errors.New("injection queue full (max 100)")
       - type Injection struct:
         - ID string
         - Text string
         - Priority bool
         - SyncChan chan struct{} (nil for async, created for sync)
       - type Queue struct:
         - mu sync.Mutex
         - items []*Injection
       - func NewQueue() *Queue
       - func (q *Queue) Enqueue(text string, priority bool) (string, error):
         - Lock, check len >= MaxQueueSize -> ErrQueueFull
         - Create Injection with uuid.New().String()
         - If priority: prepend to items slice
         - Else: append to items slice
         - Return ID, nil
       - func (q *Queue) EnqueueWithChan(text string, priority bool) (*Injection, error):
         - Same as Enqueue but also creates SyncChan: make(chan struct{})
         - Returns the full Injection pointer
       - func (q *Queue) Dequeue() *Injection:
         - Lock, if empty return nil
         - Pop first item, return it
       - func (q *Queue) Len() int -- lock, return len(items)
       - func (q *Queue) Clear() int:
         - Lock, count = len(items), set items to nil, return count
         - Close any SyncChan on cleared items (so sync callers unblock)

    - Package bridge
    - type BridgeConfig struct:
      - Command []string (the agent command + args)
      - Port int
      - Host string
      - BusyPattern string (custom override, empty = use built-in)
      - Timeout int (seconds for sync injection)
      - Verbose bool
      - Paranoid bool
      - InjectDelay int (ms)
    - type Bridge struct:
      - config BridgeConfig
      - pty *PTY
      - detector *BusyDetector
      - queue *Queue
      - injectCh chan struct{} (capacity 1)
      - ctx context.Context
      - cancel context.CancelFunc
      - startTime time.Time
      - childRunning atomic.Bool
    - func NewBridge(cfg BridgeConfig) *Bridge:
      - Creates context with cancel
      - Resolves busy pattern: if cfg.BusyPattern != "", compile it; else try to auto-detect from command name (check if cmd contains "claude", "codex", "gemini" -> use BuiltinPatterns); else use DefaultPattern
      - Creates Queue
      - Creates injectCh with cap 1
    - func (b *Bridge) Start() error:
      - Build exec.Command from config.Command
      - Set SysProcAttr with Setpgid: true
      - Create PTY with NewPTY(cmd, paranoid, injectDelay, onOutput callback)
      - onOutput callback: calls b.detector.ProcessLine(line)
      - Create BusyDetector with resolved pattern and onIdle callback
      - onIdle callback: calls b.NotifyEnqueue()
      - Start PTY (pty.Start(ctx))
      - Set childRunning = true, startTime = time.Now()
      - Go: b.detector.Run(ctx) -- goroutine for ticker
      - Go: b.injectionLoop(ctx) -- goroutine for injection
      - Go: waitForChild() -- goroutine that waits for child exit, sets childRunning=false, cancels context
    - func (b *Bridge) NotifyEnqueue():
      - Non-blocking send on injectCh (select with default)
    - func (b *Bridge) injectionLoop(ctx context.Context):
      - for/select on ctx.Done and injectCh
      - When signaled: while detector.IsIdle(), dequeue item, if nil break
      - Call pty.InjectText(item.Text)
      - If item.SyncChan != nil: close(item.SyncChan) to unblock sync caller
      - Log injection in verbose mode via slog
    - func (b *Bridge) Stop():
      - Call cancel()
      - Call pty.Close()
    - Accessor methods:
      - func (b *Bridge) Queue() *Queue
      - func (b *Bridge) IsIdle() bool -- delegates to detector.IsIdle()
      - func (b *Bridge) IsChildRunning() bool -- delegates to childRunning.Load()
      - func (b *Bridge) Uptime() time.Duration
      - func (b *Bridge) Config() BridgeConfig
